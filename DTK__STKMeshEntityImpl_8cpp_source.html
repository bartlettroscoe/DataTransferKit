105 // Return the Cartesian bounding box around an entity.
106 void STKMeshEntityImpl::boundingBox( Teuchos::Tuple<double,6>& bounds ) const
107 {
108  DTK_REQUIRE( Teuchos::nonnull(d_bulk_data) );
109 
110  Intrepid::FieldContainer<double> node_coords =
111  STKMeshHelpers::getEntityNodeCoordinates(
112  Teuchos::Array<stk::mesh::Entity>(1,d_extra_data->d_stk_entity),
113  *d_bulk_data );
114  DTK_CHECK( node_coords.rank() == 3 );
115  DTK_CHECK( node_coords.dimension(0) == 1 );
116 
117  double max = std::numeric_limits<double>::max();
118  bounds = Teuchos::tuple( max, max, max, -max, -max, -max );
119  Teuchos::Array<stk::mesh::Entity>::const_iterator entity_node_it;
120  for ( int n = 0; n < node_coords.dimension(1); ++n )
121  {
122  for ( int d = 0; d < node_coords.dimension(2); ++d )
123  {
124  bounds[d] = std::min( bounds[d], node_coords(0,n,d) );
125  bounds[d+3] = std::max( bounds[d+3], node_coords(0,n,d) );
126  }
127  }
128 }

131 // Determine if an entity is in the block with the given id.
132 bool STKMeshEntityImpl::inBlock( const int block_id ) const
133 {
134  DTK_REQUIRE( Teuchos::nonnull(d_bulk_data) );
135  const stk::mesh::PartVector& all_parts =
136  d_bulk_data->mesh_meta_data().get_parts();
137  stk::mesh::Bucket& entity_bucket =
138  d_bulk_data->bucket( d_extra_data->d_stk_entity );
139  for ( auto part_it = all_parts.begin();
140  part_it != all_parts.end();
141  ++part_it )
142  {
143  if ( Teuchos::as<int>((*part_it)->mesh_meta_data_ordinal()) == block_id )
144  {
145  return entity_bucket.member( **part_it );
146  }
147  }
148  return false;
149 }
