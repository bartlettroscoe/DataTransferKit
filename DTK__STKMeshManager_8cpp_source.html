128 //---------------------------------------------------------------------------//
129 // Create the function space.
130 void STKMeshManager::createFunctionSpace(
131  const Teuchos::RCP<stk::mesh::BulkData>& bulk_data,
132  const BasisType basis_type,
133  const Teuchos::RCP<EntitySelector>& entity_selector )
134 {
135  Teuchos::RCP<EntitySet> entity_set =
136  Teuchos::rcp( new STKMeshEntitySet(bulk_data) );
137 
138  Teuchos::RCP<EntityLocalMap> local_map =
139  Teuchos::rcp( new STKMeshEntityLocalMap(bulk_data) );
140 
141  Teuchos::RCP<EntityShapeFunction> shape_function;
142  switch( basis_type )
143  {
144  case BASIS_TYPE_GRADIENT:
145  shape_function =
146  Teuchos::rcp( new STKMeshNodalShapeFunction(bulk_data) );
147  break;
148 
149  default:
150  bool bad_basis_type = true;
151  DTK_INSIST( !bad_basis_type );
152  break;
153  }
154  DTK_CHECK( Teuchos::nonnull(shape_function) );
155 
156  d_function_space = Teuchos::rcp(
157  new FunctionSpace(entity_set,entity_selector,local_map,shape_function) );
158 }
